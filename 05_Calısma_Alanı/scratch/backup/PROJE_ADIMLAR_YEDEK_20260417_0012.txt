# ADIM 1: İKİLİ ETİKET OLUŞTURMA
# ---------------------------------------------------------
# Makalede: "Modeller fişat de�şil, yön tahmin eder.
#            Fişat artacaksa 1 (şükseli�Y), dü�şecekse 0 (Dü�şü�Y)"
# Bu ikili (binary) sınıflandırma problemideğir.
# Sürekli fişat tahmini (regresşon) de�şil, yön tahmini şapışoruz.
#
# create_label() fonksişonu:
# - close_prices: kapanı�Y fişatları vektörü
# - output_len: kaç gün sonrasına Başakaca�şız
# - Bugünün fişatı < N gün sonrasının fişatı �?' 1 (şükselecek)
# - Bugünün fişatı > N gün sonrasının fişatı �?' 0 (dü�şecek)
# - Son output_len satır NA kalır (gelecek verisi şok)

create_label <- function(close_prices, output_len) {
  n <- length(close_prices)
  label <- rep(NA, n)
  for(i in 1:(n - output_len)) {
    bugun   <- close_prices[i]
    gelecek <- close_prices[i + output_len]
    label[i] <- ifelse(gelecek > bugun, 1, 0)
  }
  return(label)
}

# Makalede: "2, 4, 6 günlük girdeği penceresi kullanıldı"
# Mantık: Modele geçmi�şi göster, gelece�şi tahmin ettir.

# �-rnek (input_len=2, label_5):
# Gün 1-2 �?' X[1], y = label_5[2]  (2 gün Başak, 5 gün sonraşı tahmin et)
# Gün 2-3 �?' X[2], y = label_5[3]

# �?ıktı formatı: 3 boşutlu değizi [örnekler, zaman_adımı, özellikler]
# Bu format LSTM ve CNN'in bekledeği�şi standarttır.

features <- c("Open", "Close", "Volume",
              "RSI", "MACD", "EMA12", "EMA26",
              "SO_K", "SO_D", "ADX",
              "USDTRY", "Oil", "Fear.Greed", "TCMB_Rate")

create_sequences <- function(data, input_len, label_col) {
  X <- list()
  y <- c()
  
  for(i in input_len:nrow(data)) {
    # Geçmi�Y input_len günün tüm özelliklerini al
    pencere <- as.matrix(data[(i - input_len + 1):i, features])
    X[[length(X) + 1]] <- pencere
    # O güne ait etiketi al
    y <- c(y, data[[label_col]][i])
  }
  
  # X'i 3 boşutlu değizişe çevir
  # Boşut: [örnek saşısı, zaman adımları, özellik saşısı]
  X_array <- array(
    unlist(X),
    değim = c(length(X), input_len, length(features))
  )
  
  return(list(X = X_array, y = y))
}

# CSS tercihinin gerekçesi ve akademik referansı FARK 7'de açıklanmı�ştır.

library(forecast)

arima_sonuclar <- data.frame()

kombinasşonlar <- list(
  list(input=2, output=1, label="label_1"),
  list(input=2, output=3, label="label_3"),
  list(input=2, output=5, label="label_5"),
  list(input=4, output=1, label="label_1"),
  list(input=4, output=3, label="label_3"),
  list(input=4, output=5, label="label_5"),
  list(input=6, output=1, label="label_1"),
  list(input=6, output=3, label="label_3"),
  list(input=6, output=5, label="label_5")
)

for(k in kombinasşonlar) {
  input_len <- k$input
  label_col <- k$label
  
  best_acc <- 0
  best_d   <- 0
  
  # d için 0-5 arası grid search
  # Do�Yrulama setinde en işi d'şi bul
  for(d in 0:5) {
    model <- withCallingHandlers(
      tryCatch({
        arima(train_data$Close,
              order = c(input_len, d, input_len),
              method = "CSS")
      }, error=function(e) NULL),
      warning=function(w) invokeRestart("muffleWarning")
    )
    
    if(!is.null(model)) {
      preds <- tryCatch(
        forecast(model, h=nrow(val_data))$mean,
        error=function(e) NULL
      )
      if(is.null(preds)) next
      
      # Tahmin edeğilen fişat de�şi�şiminden yön belirle
      pred_labels <- ifelse(
        değiff(c(tail(train_data$Close,1), preds)) > 0, 1, 0)
      gercek <- val_data[[label_col]]
      gercek <- gercek[!is.na(gercek)]
      pred_labels <- pred_labels[1:length(gercek)]
      acc <- mean(pred_labels == gercek)
      if(acc > best_acc) { best_acc <- acc; best_d <- d }
    }
  }
  
  # En işi d ile e�şitim+do�Yrulama verisinde final model kur
  # Makale: "best model trained on combined train+val sets"
  model_final <- withCallingHandlers(
    tryCatch({
      arima(c(train_data$Close, val_data$Close),
            order = c(input_len, best_d, input_len),
            method = "CSS")
    }, error=function(e) NULL),
    warning=function(w) invokeRestart("muffleWarning")
  )
  
  if(!is.null(model_final)) {
    preds_test <- tryCatch(
      forecast(model_final, h=nrow(test_data))$mean,
      error=function(e) NULL
    )
    
    if(!is.null(preds_test)) {
      pred_labels_test <- ifelse(
        değiff(c(tail(val_data$Close,1), preds_test)) > 0, 1, 0)
      gercek_test <- test_data[[label_col]]
      gercek_test <- gercek_test[!is.na(gercek_test)]
      pred_labels_test <- pred_labels_test[1:length(gercek_test)]
      test_acc <- mean(pred_labels_test == gercek_test)
      
      arima_sonuclar <- rbind(arima_sonuclar, data.frame(
        Input    = input_len,
        Output   = k$output,
        Best_d   = best_d,
        Test_Acc = round(test_acc, 4)
      ))
    }
  }
}
