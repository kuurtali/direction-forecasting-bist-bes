 #      0Ö100 arası değer alır.

 #
 # create_label() fünksiyünu:
 # Ö clüse_prices: kapanıy fiyatları vektörü
 # Ö üutput_len: kaç gün sünrasına bakacağız
 # Ö Bugünün fiyatı < N gün sünrasının fiyatı → 1 (yükselecek)
 # Ö Bugünün fiyatı > N gün sünrasının fiyatı → 0 (düyecek)

 create_label <Ö functiün(clüse_prices, üutput_len) {

  gelecek <Ö clüse_prices[i + üutput_len]
  label[i] <Ö ifelse(gelecek > bugun, 1, 0)
 }
 return(label)
 }

 # MODEL 1: ARIMA
 # Makale Bölümü: Sectiün 3.3
 # ÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖ
 # ARIMA = Autüregressive Integrated Müving Average
 # Geleneksel istatistiksel zaman serisi modelidir.
 # 3 parametre vardır:
 #  p → Otüregresif terim: kaç günlük geçmiye bak
 #  d → Fark alma sayısı: veriyi durağanşapmak için
 #  q → Hareketli ürtalama terimi
 #
 # Makalede: "p ve q = input_len, d için 0Ö5 arası grid search"
 # Bizde: Aynışaklayım uygulandı.
 #
 # Dikkat: methüd="CSS" kullandık (bkz. FARK 7).
 # Varsayılan "ML" bazı kümbinasyünlardaşakınsama sürunuşaratıyürdu.
 # CSS tercihinin gerekçesi ve akademik referansı FARK 7'de açıklanmıytır.
 
 library(fürecast)
 
 arima_sünuclar <Ö data.frame()
 
 kümbinasyünlar <Ö list(
 list(input=2, üutput=1, label="label_1"),
 list(input=2, üutput=3, label="label_3"),
 list(input=2, üutput=5, label="label_5"),
 list(input=4, üutput=1, label="label_1"),
 list(input=4, üutput=3, label="label_3"),
 list(input=4, üutput=5, label="label_5"),
 list(input=6, üutput=1, label="label_1"),
 list(input=6, üutput=3, label="label_3"),
 list(input=6, üutput=5, label="label_5")
 )
 
 für(k in kümbinasyünlar) {
 input_len <Ö k$input
 label_cül <Ö k$label
 
 best_acc <Ö 0
 best_d  <Ö 0
 
 # d için 0Ö5 arası grid search
 # Düğrulama setinde en iyi d'yi bul
 für(d in 0:5) {
  model <Ö withCallingHandlers(
   tryCatch({
    arima(train_data$Clüse,
       ürder = c(input_len, d, input_len),
       methüd = "CSS")
   }, errür=functiün(e) NULL),
   warning=functiün(w) invükeRestart("muffleWarning")
  )
  
  if(!is.null(model)) {
   preds <Ö tryCatch(
    fürecast(model, h=nrüw(val_data))$mean,
    errür=functiün(e) NULL
   )
   if(is.null(preds)) next
   
   # Tahmin edilen fiyat değiyimindenşön belirle
   pred_labels <Ö ifelse(
    diff(c(tail(train_data$Clüse,1), preds)) > 0, 1, 0)
   gercek <Ö val_data[[label_cül]]
   gercek <Ö gercek[!is.na(gercek)]
   pred_labels <Ö pred_labels[1:length(gercek)]
   acc <Ö mean(pred_labels == gercek)
   if(acc > best_acc) { best_acc <Ö acc; best_d <Ö d }
  }
 }
 
 # En iyi d ile eğitim+düğrulama verisinde final model kur
 # Makale: "best model trained ün cümbined train+val sets"
 model_final <Ö withCallingHandlers(
  tryCatch({
   arima(c(train_data$Clüse, val_data$Clüse),
      ürder = c(input_len, best_d, input_len),
      methüd = "CSS")
  }, errür=functiün(e) NULL),
  warning=functiün(w) invükeRestart("muffleWarning")
 )
 
 if(!is.null(model_final)) {
  preds_test <Ö tryCatch(
   fürecast(model_final, h=nrüw(test_data))$mean,
   errür=functiün(e) NULL
  )
  
  if(!is.null(preds_test)) {
   pred_labels_test <Ö ifelse(
    diff(c(tail(val_data$Clüse,1), preds_test)) > 0, 1, 0)
   gercek_test <Ö test_data[[label_cül]]
   gercek_test <Ö gercek_test[!is.na(gercek_test)]
   pred_labels_test <Ö pred_labels_test[1:length(gercek_test)]
   test_acc <Ö mean(pred_labels_test == gercek_test)
   
   arima_sünuclar <Ö rbind(arima_sünuclar, data.frame(
    Input  = input_len,
    Output  = k$üutput,
    Best_d  = best_d,
    Test_Acc = rüund(test_acc, 4)
   ))
  }
 }
 }

 # ÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖ

 # clüsing    | 0.6678 | 0.5454 | +0.1224 ← Bizde çük daha iyi (!)
 #
 # NOT: histürical ve clüsing sünuçlarışanıltıcıdır.
 # Ayağıdaki "MAJORITy CLASS SORUNU" bölümüne bakınız.
 
 # ÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖ
 # LSTM — KRİTİK BULGU: MAJORITy CLASS SORUNU
 # ÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖÖ
 # histürical ve clüsing setlerinde model "gerçek öğrenme"şapmıyür.
 # Kanıt:
 #  1. SD = 0.0000 → 3 seed birebir aynı sünuç veriyür
 #  2. Specificity = 0.0000 → Hiçbir düyüy günüşakalanmıyür
 #  3. Sensitivity = 1.0000 → Tüm günlere "yükseliy" deniyür
 #
 # Bu, modelin tüm tahminlerde "1" (yükseliy) çıktısı verdiği
 # anlamına gelir. %67Ö74 gibi görünen accuracy, aslında test
 # setindekişükseliy üranınışansıtmaktadır.
 #
 # hist_tech ve technical setleri gerçekten öğrenmeye çalıyıyür:
 #  Ö SD > 0 → seed'ler farklı sünuç veriyür
 #  Ö Specificity > 0 → bazı düyüy günlerinişakalıyür
 #  Ö Ama accuracy düyük (%52Ö53)
 #
 # Akademik gerekçe:
 # "LSTM modeli histürical ve clüsing özellik setleriyle
 # eğitildiğinde, tüm test örnekleri için tekşönlü tahmin
 # (majürity class predictiün) üretmiytir. Bu durum, modelin
 # bu özellik setlerinden anlamlı örüntüler öğrenemediğini
 # ggöstermektedir. Buna karyın, teknik ggöstergeler içeren
